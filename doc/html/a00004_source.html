55 /** RTC clock freq from LF clock in Hertz. Changable, but should be usable by @ref RTC_TICKS */
56 #define RTC_FREQUENCY 32768
57 /* Check if RTC_FREQUENCY is power of 2 and less than or equal to 32.768 kHz */
58 #if (!(!(RTC_FREQUENCY & (RTC_FREQUENCY-1)) && RTC_FREQUENCY && (RTC_FREQUENCY<=32768)))
59 #error RTC_FREQUENCY must be a power of 2 with a maximum frequency of 32768 Hz
60 #endif
61 /** Prescalar to the LF Clock for the RTC peripheral based on f = LFCLK/(prescaler + 1) */
62 #define RTC_PRESCALER ((LFCLK_FREQUENCY/RTC_FREQUENCY) - 1)
63 /** Calculation to findout the number of RTC ticks for the passed time in milli-seconds */
64 #define RTC_TICKS(ms) ((uint32_t) DIV_ROUNDED( (RTC_FREQUENCY*ms) , 1000) )
