55 void ms_timer_init(void){
56 
57  NRF_RTC1->TASKS_STOP = 1;
58 
59  for(uint32_t i = MS_TIMER0; i < MS_TIMER_MAX; i++){
60  ms_timer_t[i].timer_mode_t = SINGLE_CALL;
61  ms_timer_t[i].timer_handler = NULL;
62  }
63 
64  ms_timers_status = 0;
65  NRF_RTC1->PRESCALER = RTC_PRESCALER; // Set prescaler to a TICK of RTC_FREQUENCY.
66  NVIC_SetPriority(RTC1_IRQn,PRIORITY_RTC1_IRQn);
67  NVIC_EnableIRQ(RTC1_IRQn); // Enable Interrupt for RTC1 in the core.
68 }

70 /**@todo Take care of values of ticks passed which are greater than 2^24, now it is suppressed to 2^24 when greater
71  * @warning Works only for input less than 512000 milli-seconds, or 8.5 min without when @ref RTC_PRESCALER is 0 */
72 void start_ms_timer(timer_num id, timer_mode mode, uint32_t ticks, void (*handler)(void)){
73 
74  /* make sure the number of ticks to interrupt is less than 2^24 */
75  ticks &= 0xFFFFFF;
76 
77  ms_timer_t[id].timer_handler = handler;
78  if(mode == SINGLE_CALL){
79  ms_timer_t[id].timer_mode_t = SINGLE_CALL;
80  }else{
81  ms_timer_t[id].timer_mode_t = ticks;
82  }
83 
84  NRF_RTC1->CC[id] = NRF_RTC1->COUNTER + ticks;
85 
86  NRF_RTC1->EVENTS_COMPARE[id] = 0;
87  NRF_RTC1->EVTENSET = 1 << (RTC_INTENSET_COMPARE0_Pos + id);
88  NRF_RTC1->INTENSET = 1 << (RTC_INTENSET_COMPARE0_Pos + id);
89 
90  if(ms_timers_status == 0){
91  lfclk_init();
92  NRF_RTC1->TASKS_START = 1;
93  }
94  ms_timers_status |= 1 << id;
95 }
96 
97 void stop_ms_timer(timer_num id){
98  ms_timer_t[id].timer_mode_t = SINGLE_CALL;
99  ms_timer_t[id].timer_handler = NULL;
100  ms_timers_status &= ~(1 << id);
101  NRF_RTC1->EVTENCLR = 1 << (RTC_INTENSET_COMPARE0_Pos + id);
102  NRF_RTC1->INTENCLR = 1 << (RTC_INTENSET_COMPARE0_Pos + id);
103 
104  if(ms_timers_status == 0){
105  NRF_RTC1->TASKS_STOP = 1;
106  lfclk_deinit();
107  }
108 }

111  if((ms_timers_status & (1<<id)) == 0){
112  return false;
113  }
114  else{
115  return true;

119 /** @brief Function for handling the RTC1 interrupts.
120  * Triggered Compare register of timer ID
121  */
122 void
123 RTC1_IRQHandler(){
124  for(uint32_t i = MS_TIMER0; i < MS_TIMER_MAX; i++){
125  if(NRF_RTC1->EVENTS_COMPARE[i]){
126  NRF_RTC1->EVENTS_COMPARE[i] = 0;
127 
128  if(ms_timer_t[i].timer_handler != NULL){
129  ms_timer_t[i].timer_handler();
130  }
131 
132  if(ms_timer_t[i].timer_mode_t == SINGLE_CALL){
133  stop_ms_timer(i);
134  }else{
135  NRF_RTC1->CC[i] += ms_timer_t[i].timer_mode_t;
136  }
137  }
138  }
