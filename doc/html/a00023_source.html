192 static void collect_packet(void){
193  /* Collect a max of LOGGER_NUMBER packets, otherwise reduce observer window */
194  if(packet_count < LOGGER_NUMBER){
195  uint32_t time = read_time_us();
196  memcpy(*(packets + packet_count), pdu, pdu[1]+2);
197  memcpy(*(packets + packet_count) + MAX_PDU_SIZE, &time, 4); //sizeof(time)=4
198  packets[packet_count][MAX_PDU_SIZE+4] = NRF_RADIO->DATAWHITEIV - 64; //6th bit is always 1
199  if(NRF_RADIO->CRCSTATUS == 1){
200  packets[packet_count][MAX_PDU_SIZE+4] |= 0x80;
201  }
202  packets[packet_count][MAX_PDU_SIZE+5] = rssi;
203  packet_count++;
204  }
205 }

226  case TYPE_UUID32_INC:

287  * 1 LE General Discoverable Mode\n
288  * 2 BR/EDR Not Supported\n
289  * 3 Simultaneous LE and BR/EDR to Same Device is Capable (Controller)\n

293  case TYPE_FLAGS:
294  case TYPE_SERVICE_SOLICITATION16:
295  case TYPE_SERVICE_SOLICITATION128:
296  case TYPE_APPEARANCE:
297  default:

302  }

308 static void print_collected(uint8_t * packet_to_print){
309  uint32_t packet_time;
310  memcpy(&packet_time,packet_to_print + MAX_PDU_SIZE,4);
311  printf("Time:");
312  printfcomma(packet_time);
313  printf("us|");
314  printf("Channel:%d|", (int) packet_to_print[MAX_PDU_SIZE+4]&0x7F);
315  printf("RSSI:%d|", (int) packet_to_print[MAX_PDU_SIZE+5]);
316  printf("CRC:%s|", (packet_to_print[MAX_PDU_SIZE+4]&0x80) ? "Y" : "N");
317 
318  /* 0000 ADV_IND
319  0001 ADV_DIRECT_IND
320  0010 ADV_NONCONN_IND
321  0011 SCAN_REQ
322  0100 SCAN_RSP
323  0101 CONNECT_REQ
324  0110 ADV_SCAN_IND
325  0111-1111 Reserved */
326 // printf("Type:P%d ",(int) (packet_to_print[0]) & 0xF);
327  printf("Type:");
328  switch((packet_to_print[0]) & 0xF){
329  case TYPE_ADV_IND: printf("ADV_IND|");
330  break;
331  case TYPE_ADV_DIRECT_IND: printf("ADV_DIRECT_IND|");
332  break;
333  case TYPE_ADV_NONCONN_IND: printf("ADV_NONCONN_IND|");
334  break;
335  case TYPE_SCAN_REQ: printf("SCAN_REQ|");
336  break;
337  case TYPE_SCAN_RSP: printf("SCAN_RSP|");
338  break;
339  case TYPE_CONNECT_REQ: printf("CONNECT_REQ|");
340  break;
341  case TYPE_ADV_SCAN_IND: printf("ADV_SCAN_IND|");
342  break;
343  default:printf("Reserved|");
344  }
345 
346  /* 1: Random Tx address, 0: Public Tx address */
347  printf("TxAdrs:%s|",(int) (packet_to_print[0] & 0x40)?"Random":"Public");
348  /* 1: Random Rx address, 0: Public Rx address */
349  printf("RxAdrs:%s|",(int) (packet_to_print[0] & 0x80)?"Random":"Public");
350  /* Length of the data packet */
351  printf("Len:%d|",(int) packet_to_print[1] & 0x3F);
352 
353  uint8_t i;
354  uint8_t f = packet_to_print[1] + 2;
355  printf("Adrs:");
356  for(i = 7; i>2; i--){
357  printf("%02X:", packet_to_print[i]);
358  }printf("%02X", packet_to_print[2]);
359  printf("\n");
360 
361  if(packet_to_print[MAX_PDU_SIZE+4]&0x80){
362  printf("Packet:");
363  for (i = 0; i < f; i++) {
364  printf("%02X ", packet_to_print[i]);
365  }
366  printf("\n");
367  uint32_t type_temp = (packet_to_print[0] & 0x7);
368  switch(type_temp){
369  case TYPE_ADV_IND:
370  case TYPE_ADV_NONCONN_IND:
371  case TYPE_SCAN_RSP:
372  case TYPE_ADV_SCAN_IND:
373  adv_seg_print(packet_to_print+8,f-8);
374  printf("\n");
375  break;
376  default: break;
377  }
378  }else{
379  printf("Data corrupted. CRC check failed.\n");
380  }
381 }

383 void obsvr_radio_init(){
384 
385  /* Refresh all the registers in RADIO */
386  NRF_RADIO->POWER = 0;
387  NRF_RADIO->POWER = 1;
388 
389  /* Set RADIO mode to Bluetooth Low Energy. */
390  NRF_RADIO->MODE = RADIO_MODE_MODE_Ble_1Mbit << RADIO_MODE_MODE_Pos;
391 
392  /* Copy the BLE override registers from FICR */
393  NRF_RADIO->OVERRIDE0 = NRF_FICR->BLE_1MBIT[0];
394  NRF_RADIO->OVERRIDE1 = NRF_FICR->BLE_1MBIT[1];
395  NRF_RADIO->OVERRIDE2 = NRF_FICR->BLE_1MBIT[2];
396  NRF_RADIO->OVERRIDE3 = NRF_FICR->BLE_1MBIT[3];
397  NRF_RADIO->OVERRIDE4 = NRF_FICR->BLE_1MBIT[4];
398 
399 
400  /* Set access address to 0x8E89BED6. This is the access address to be used
401  * when send packets in obsvrertise channels.
402  *
403  * Since the access address is 4 bytes long and the prefix is 1 byte long,
404  * we first set the base address length to be 3 bytes long.
405  *
406  * Then we split the full access address in:
407  * 1. Prefix0: 0x0000008E (LSB -> Logic address 0)
408  * 2. Base0: 0x89BED600 (3 MSB)
409  *
410  * At last, we enable reception for this address.
411  */
412  NRF_RADIO->PCNF1 |= 3UL << RADIO_PCNF1_BALEN_Pos;
413  NRF_RADIO->BASE0 = 0x89BED600;
414  NRF_RADIO->PREFIX0 = 0x0000008E;
415  NRF_RADIO->RXADDRESSES = 0x00000001;
416 
417  /* Enable data whitening. */
418  NRF_RADIO->PCNF1 |= RADIO_PCNF1_WHITEEN_Enabled << RADIO_PCNF1_WHITEEN_Pos;
419 
420  /* Set maximum PAYLOAD size. */
421  NRF_RADIO->PCNF1 |= MAX_PDU_SIZE << RADIO_PCNF1_MAXLEN_Pos;
422 
423  /* Configure CRC.
424  *
425  * First, we set the length of CRC field to 3 bytes long and ignore the
426  * access address in the CRC calculation.
427  *
428  * Then we set CRC initial value to 0x555555.
429  *
430  * The last step is to set the CRC polynomial to
431  * x^24 + x^10 + x^9 + x^6 + x^4 + x^3 + x + 1.
432  */
433  NRF_RADIO->CRCCNF = RADIO_CRCCNF_LEN_Three << RADIO_CRCCNF_LEN_Pos |
434  RADIO_CRCCNF_SKIP_ADDR_Skip
435  << RADIO_CRCCNF_SKIP_ADDR_Pos;
436  NRF_RADIO->CRCINIT = 0x555555UL;
437  NRF_RADIO->CRCPOLY = SET_BIT(24) | SET_BIT(10) | SET_BIT(9) |
438  SET_BIT(6) | SET_BIT(4) | SET_BIT(3) |
439  SET_BIT(1) | SET_BIT(0);
440 
441  /* Configure header size.
442  *
443  * The Advertise has the following format:
444  * RxAdd(1b) | TxAdd(1b) | RFU(2b) | PDU Type(4b) | RFU(2b) | Length(6b)
445  *
446  * And the nRF51822 RADIO packet has the following format
447  * (directly editable fields):
448  * S0 (0/1 bytes) | LENGTH ([0, 8] bits) | S1 ([0, 8] bits)
449  *
450  * We can match those fields with the Link Layer fields:
451  * S0 (1 byte) --> PDU Type(4bits)|RFU(2bits)|TxAdd(1bit)|RxAdd(1bit)
452  * LENGTH (6 bits) --> Length(6bits)
453  * S1 (0 bits) --> S1(0bits)
454  */
455  NRF_RADIO->PCNF0 |= (1 << RADIO_PCNF0_S0LEN_Pos) | /* 1 byte */
456  (8 << RADIO_PCNF0_LFLEN_Pos) | /* 6 bits */
457  (0 << RADIO_PCNF0_S1LEN_Pos); /* 2 bits */
458 
459  /* Set the pointer to write the incoming packet. */
460  NRF_RADIO->PACKETPTR = (uint32_t) pdu;
461 
462  /* Set the initial freq to obsvr as channel 37 */
463  NRF_RADIO->FREQUENCY = 80;//ADV_CHANNEL[3];
464 
465  /* Configure the shorts for observing
466  * READY event and START task
467  * ADDRESS event and RSSISTART task
468  * END event and START task
469  * */
470  NRF_RADIO->SHORTS = RADIO_SHORTS_READY_START_Msk |
471  RADIO_SHORTS_ADDRESS_RSSISTART_Msk |
472  RADIO_SHORTS_END_START_Msk;
473 
474  NRF_RADIO->INTENSET = RADIO_INTENSET_DISABLED_Msk |
475  RADIO_INTENSET_RSSIEND_Msk |
476  RADIO_INTENSET_END_Msk;
477 
478  // Enable Interrupt for RADIO in the core.
479  NVIC_SetPriority(RADIO_IRQn, PRIORITY_RADIO_IRQn);
480  NVIC_EnableIRQ(RADIO_IRQn);
481 }

543  * @dot
544  * digraph uart_handler_flow {
545  * a [shape = oval, label = "Called with unsigned char pointer as argument"]
546  * b [shape = diamond, label ="Is the received string 'START?"]
547  * c [shape = diamond, label ="Is the received string 'STOP?"]
548  * d [shape = diamond, label ="Is the Scan-Window Timer running?"]
549  * g [shape = box, label ="Start the repetitive timer with duration of scan window "]
550  * e [shape = box, label ="1. Disable the Radio\n2. Stop the Scan-Window Timer\n3. Stop the Scan-Interval Timer"]
551  * f [shape = oval, label ="Return"]
552  * a -> b;
553  * b -> c [label ="false"];
554  * b -> d [label ="true"];
555  * c -> e [label ="true"];
556  * c -> f [label ="false"];
557  * d -> g [label ="false"];
558  * d -> f [label ="true"];
559  * e -> f;
560  * g -> f;
561  * }
562  * @enddot

564 static void obsvr_req(uint8_t * ptr){
565 
566  /* Check if the string starts with 'ST' */
567  if(strncmp((const char *)ptr,"ST",2)==0){
568  /* if 'START' is received, start the scan_window repetitive timer */
569  if(strcmp((const char *)ptr+2,"ART")==0){
570  if(is_ms_timer_on(MS_TIMER0) == false){
571  start_ms_timer(MS_TIMER0, REPEATED_CALL, RTC_TICKS(SCAN_INTERVAL), &obsvr_interval);
572  }
573  }
574  /* if 'STOP' is received, disable the Radio and stop both the timers*/
575  else if(strcmp((const char *)ptr+2,"OP")==0){
576  NRF_RADIO->EVENTS_DISABLED = 0UL;
577  NRF_RADIO->TASKS_DISABLE = 1UL;
578  while (NRF_RADIO->EVENTS_DISABLED == 0UL);
579 
580  stop_ms_timer(MS_TIMER0);
581  stop_ms_timer(MS_TIMER1);
582  }
583  }
