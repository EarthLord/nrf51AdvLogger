55 void rx_collect(uint8_t rx_data){
56  static uint32_t count = 0;
57  if(rx_data != LINE_END){
58  if(count < BUFFER_SIZE -1){
59  rx_buffer[count] = rx_data;
60  count++;
61  }
62  }else{
63  rx_buffer[count] = '\0';
64  if(rx_handler != NULL){
65  rx_handler(rx_buffer);
66  }
67  count = 0;
68  }
69 }

71 /**
72  * UART interrupt routine.

75 void UART0_IRQHandler(void)
76 {
77  /* Waits for RX data to be received, but
78  * no waiting actually since RX causes interrupt. */
79  while(NRF_UART0->EVENTS_RXDRDY != 1);
80  NRF_UART0->EVENTS_RXDRDY = 0;

112 void uart_init(){
113  /* Make rx_handler NULL, configure it with set_rx_handler */
114  rx_handler = NULL;
115 
116  /* Configure TX and RX pins from board.h */
117  nrf_gpio_cfg_output(TX_PIN_NUMBER);
118  nrf_gpio_cfg_input(RX_PIN_NUMBER, NRF_GPIO_PIN_NOPULL);
119  NRF_UART0->PSELTXD = TX_PIN_NUMBER;
120  NRF_UART0->PSELRXD = RX_PIN_NUMBER;
121 
122  /* Configure CTS and RTS pins if HWFC is true in board.h */
123  if(HWFC){
124  nrf_gpio_cfg_output(RTS_PIN_NUMBER);
125  nrf_gpio_cfg_input(CTS_PIN_NUMBER, NRF_GPIO_PIN_NOPULL);
126  NRF_UART0->PSELRTS = RTS_PIN_NUMBER;
127  NRF_UART0->PSELCTS = CTS_PIN_NUMBER;
128  NRF_UART0->CONFIG = (UART_CONFIG_HWFC_Enabled << UART_CONFIG_HWFC_Pos);
129  }
130 
131  /* Configure other UART parameters, BAUD rate is defined in uart_nrf.h */
132  NRF_UART0->BAUDRATE = (UART_BAUDRATE << UART_BAUDRATE_BAUDRATE_Pos);
133  NRF_UART0->ENABLE = (UART_ENABLE_ENABLE_Enabled << UART_ENABLE_ENABLE_Pos);
134  NRF_UART0->EVENTS_RXDRDY = 0;
135 
136  // Enable UART RX interrupt only
137  NRF_UART0->INTENSET = (UART_INTENSET_RXDRDY_Set << UART_INTENSET_RXDRDY_Pos);
138 
139  NVIC_SetPriority(UART0_IRQn, PRIORITY_UART0_IRQn);
140  NVIC_EnableIRQ(UART0_IRQn);
141 
142  /* Start reception and transmission */
143  NRF_UART0->TASKS_STARTTX = 1;
144  NRF_UART0->TASKS_STARTRX = 1;
145 }
